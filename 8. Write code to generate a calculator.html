Q. Write code to generate a calculator.



num1 = float(input("Enter first number: "))
num2 = float(input("Enter second number: "))

while True :
	operation = input("Select operation (+, -, *, /): ")
	if operation in "+-*/" :
		break
	else:
		print("The symbol you entered is invalid. Please re try.")

if operation == '+':
    print(num1 + num2)
elif operation == '-':
    print(num1 - num2)
elif operation == '*':
    print(num1 * num2)
elif operation == '/':
    print(num1 / num2)
